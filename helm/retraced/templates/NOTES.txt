API Service ({{ .Values.api.service.type }}):
{{- if eq .Values.api.service.type "LoadBalancer" }}
  It may take a few minutes for the LoadBalancer IP to be available.
  Watch: kubectl get svc --namespace {{ .Release.Namespace }} {{ include "retraced.fullname" . }}-api -w

  Once available:
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "retraced.fullname" . }}-api \
    -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "API: http://$SERVICE_IP:{{ .Values.api.service.port }}"
{{- else if eq .Values.api.service.type "ClusterIP" }}
  kubectl port-forward --namespace {{ .Release.Namespace }} \
    svc/{{ include "retraced.fullname" . }}-api {{ .Values.api.service.port }}:{{ .Values.api.service.port }}
  echo "API: http://127.0.0.1:{{ .Values.api.service.port }}"
{{- else if eq .Values.api.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "retraced.fullname" . }}-api \
    -o jsonpath='{.spec.ports[0].nodePort}')
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} \
    -o jsonpath='{.items[0].status.addresses[0].address}')
  echo "API: http://$NODE_IP:$NODE_PORT"
{{- end }}
